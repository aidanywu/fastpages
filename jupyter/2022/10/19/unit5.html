public int scoreGuess (String guess){
    int count = 0;
    for (int i = 0; i <= secret.length() - guess.length(); i++) {
        if (secret.substring(i, i + guess.length()). equals(guess)) {
            count++;
        }
    }
    return count * Math.pow(guess.length(), 2);
}
